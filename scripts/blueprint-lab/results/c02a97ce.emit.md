### Injectable git state (the seam the primitive exposes via `RunSweepMeasurementOpts`, `sweep-measurement.ts:43-47`)
Build a **mutable** git model so idempotence is testable, not a static probe:
```ts
function makeGitState(aheadByBranch: Record<string, number>) {
  const live = new Set(Object.keys(aheadByBranch));
  const probe: GitProbe = (branch) =>
    live.has(branch)
      ? { exists: true, ahead: aheadByBranch[branch], behind: 0, mergeable: true }
      : { exists: false, ahead: null, behind: null, mergeable: null };
  const deleted: string[] = [];
  const runner: BranchGcRunner = {
    revParse: (b) => (live.has(b) ? `sha-${b}` : null),
    deleteBranch: (b) => { live.delete(b); deleted.push(b); return true; },
  };
  return { probe, runner, deleted, live };
}
```
`gcEpicBranches` only deletes `ahead===0` branches and flags `ahead>0` (verified `landed-epic-sweep.test.ts:85-124`), so this model makes GC deletions actually mutate `live`, which the post-GC re-probe (step 4, `sweep-measurement.ts:96-104`) then observes.
