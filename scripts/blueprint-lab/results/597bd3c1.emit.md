## Change 1 — throttle gate + pass wrapper in `src/services/landed-epic-sweep.ts`

Add near the top of the module (after the imports at `landed-epic-sweep.ts:12`):

```ts
/** Minimum spacing between PERIODIC landed-epic sweeps for a single project. This
 *  reconcile/GC pass is hygiene (finalize a [LAND] leaf, prune a fully-landed branch),
 *  not claim/build-latency-sensitive — same throttle shape and window as
 *  ARCHIVAL_SWEEP_INTERVAL_MS (archival-sweep.ts:20). */
export const LANDED_EPIC_SWEEP_INTERVAL_MS = 300_000; // 5 min

const lastLandedEpicSweepMs = new Map<string, number>();

/** Throttle gate: returns true (recording `now`) when the periodic sweep is due for a
 *  project, false while within LANDED_EPIC_SWEEP_INTERVAL_MS. First call always runs
 *  (boot freshness). `now` injectable for deterministic tests. Mirrors
 *  shouldRunArchivalSweep (archival-sweep.ts:30). */
export function shouldRunLandedEpicSweep(project: string, now: number = Date.now()): boolean {
  const last = lastLandedEpicSweepMs.get(project);
  if (last !== undefined && now - last < LANDED_EPIC_SWEEP_INTERVAL_MS) return false;
  lastLandedEpicSweepMs.set(project, now);
  return true;
}

/** Test seam: clear the landed-epic-sweep throttle clock. */
export function _resetLandedEpicSweepThrottle(project?: string): void {
  if (project === undefined) lastLandedEpicSweepMs.clear();
  else lastLandedEpicSweepMs.delete(project);
}
```

Add a combined pass wrapper at the end of the module (after `gcEpicBranches`, `landed-epic-sweep.ts:154`), self-gated on `force` like `runArchivalSweep` (archival-sweep.ts:47-64):

```ts
export interface LandedEpicSweepPassResult {
  reconciled: string[];
  gcDeleted: string[];
  gcFlagged: string[];
}

/** One orchestrator pass: reconcile landed converged-mission epics (finalize their
 *  [LAND] leaf) then GC fully-landed epic branches. `force` lets tests/direct callers
 *  run outside the throttle window; the periodic tick wiring passes force:true since it
 *  already consulted shouldRunLandedEpicSweep as the outer gate (same two-seam shape as
 *  runArchivalSweep). Best-effort: a GC failure never blocks reconcile. */
export async function runLandedEpicSweepPass(
  project: string,
  opts: { probe?: GitProbe; runner?: BranchGcRunner; baseRef?: string; now?: number; force?: boolean } = {},
): Promise<LandedEpicSweepPassResult> {
  const now = opts.now ?? Date.now();
  if (!opts.force && !shouldRunLandedEpicSweep(project, now)) {
    return { reconciled: [], gcDeleted: [], gcFlagged: [] };
  }
  const { reconciled } = await reconcileLandedEpics(project, { probe: opts.probe, baseRef: opts.baseRef });
  const gc = gcEpicBranches(project, { probe: opts.probe, runner: opts.runner, baseRef: opts.baseRef });
  return { reconciled, gcDeleted: gc.deleted, gcFlagged: gc.flagged };
}
```

`GitProbe`, `BranchGcRunner` are already in scope (imported at line 10 / defined at line 67).
